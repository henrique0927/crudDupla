CREATE TABLE professores (
id INT PRIMARY KEY,
nome VARCHAR(255)
);


CREATE TABLE aulas (
id INT PRIMARY KEY,
sala VARCHAR(255)
);


CREATE TABLE dia_hora (
professor_id INT,
aula_id INT,
hora_aula DATETIME,
PRIMARY KEY (professor_id, aula_id),
FOREIGN KEY (professor_id) REFERENCES professores(id),
FOREIGN KEY (aula_id) REFERENCES aulas(id)
);
